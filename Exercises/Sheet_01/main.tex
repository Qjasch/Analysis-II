\section{Young's inequality}
\begin{enumerate}
    \item Consider $h(t, p) := \frac{t}{p} + \frac{1}{q} - t^{\frac{1}{p}} 
    = \frac{t - 1}{p} + 1 - t^{\frac{1}{p}}$. 
    Then $\frac{\partial h^2}{\partial t^2} = \frac{1}{p} \cdot \frac{1}{p} t^{\frac{1}{p} - 2} = \frac{1}{p^2} t^{\frac{1}{p} - 2} > 0$, so $h$ is convex in $t$. Moreover, $h(1, p) = 0$ and $\frac{\partial h}{\partial t}(1, p) = 0$, so $h(t, p) \geq 0$ for all $t > 0$.

    \item Let $a, b > 0$ and $t := \frac{a^p}{b^q}$. Then $t > 0$ and
    \[\frac{a^p}{b^q} = t \leq \frac{t}{p} + \frac{1}{q} = \frac{a^p}{p b^q} + \frac{1}{q} = \frac{a^p}{p} + \frac{b^q}{q}.\]    
\end{enumerate}
