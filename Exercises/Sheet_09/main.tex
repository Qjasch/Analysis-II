\section{Volume}
$\text{vol}(B) = \int_{0}^{2\pi} \int_{0}^{2} \left(\sqrt{8 - r^2} - \frac{r^2}{2} \right) \, dr \, d\theta 
= \int_{0}^{2\pi} \left(\int_{0}^{\frac{\pi}{2}} \sqrt{8} \cos \varphi \sqrt{8 - \sqrt{8} \sin \varphi}\, d\varphi - \frac{4}{3}\right) \, d\theta$ \\
$ = \int_{0}^{2\pi} \left(\int_{0}^{\frac{\pi}{2}} 8 \cos^2 \varphi \, d\varphi - \frac{4}{3}\right) \, d\theta
= \int_{0}^{2\pi} \left(2 \pi - \frac{4}{3}\right) \, d\theta
= 2 \pi \left(2 \pi - \frac{4}{3}\right)$ 

\section{Multiple integrals}
\begin{enumerate}
    \item $\int_{0}^{2} \int_{0}^{1} (x^3 + 3x^2 y + y^3) \, dy \, dx
    = \int_{0}^{2} \left[x^3 y + \frac{3x^2 y^2}{2} + \frac{y^4}{4}\right]_0^1 \, dx
    = \left[\frac{x^4}{4} + \frac{x^3}{2} + \frac{x}{4}\right]_0^2
    = \frac{17}{2}$ 
    
    \item $\int_{0}^{\pi} \int_{x}^{\pi} x \cos(x + y) \, dy \, dx
    = \int_{0}^{\pi} \left[x \sin(x + y) \right]_x^{\pi} \, dx
    = \int_{0}^{\pi} x \sin(2x) \, dx
    = \frac{\pi}{2}$ 
    
    \item $\int_{1}^{2} \int_{1}^{3-x} \frac{1}{(x+y)^3} \, dy \, dx
    = \int_{1}^{2} \left[-\frac{1}{2(x+y)^2} \right]_1^{3-x} \, dx
    = \int_{1}^{2} \left(-\frac{1}{18} + \frac{1}{2(x+1)^2} \right) \, dx
    = \frac{1}{12}$ 
\end{enumerate}


\section{Fubini theorem}
$\int_{0}^{\sqrt{\pi}} \int_{x}^{\sqrt{\pi}} \sin y^2 \, dy \, dx
= \int_{0}^{\sqrt{\pi}} \int_{0}^{y} \sin y^2 \, dx \, dy
= \int_{0}^{\sqrt{\pi}} \sin y^2 \left[ x \right]_{0}^{y} \, dy
= \int_{0}^{\sqrt{\pi}} y \sin y^2 \, dy
= 1$, 

\vspace{10pt}

$\int_{-1}^{1} \int_{\abs{y}}^{1} (x + y^2) \, dx \, dy
= \int_{0}^{1} \int_{-x}^{x} (x + y)^2 \, dy \, dx
= \int_{0}^{1} \frac{(2x)^3}{3} \, dx 
= \left[\frac{8x^4}{12}\right]_0^1 
= \frac{2}{3}$

\section{Counterexample to Fubini}
$\int_{0}^{\infty} f(x, y) \, dx \, dy 
= \int_{0}^{\infty} \left(\int_{0}^{y} -e^{-x-y} \, dx + \int_{y}^{\infty} e^{y-x} \, dx \right) dy
= \int_{0}^{\infty} e^{-y} \, dy
= 1$, \\

$\int_{0}^{\infty} f(x, y) \, dy \, dx 
= \int_{0}^{\infty} \left(\int_{0}^{x} e^{y-x} \, dy + \int_{x}^{\infty} -e^{x-y} \, dy \right) dx
= \int_{0}^{\infty} -e^{-x} \, dx
= -1$

\vspace{10pt}

This is not a contradiction to Fubini since the domain is not bounded.
